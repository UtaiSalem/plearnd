```php
// app/Http/Controllers/Learn/Student/HomeVisit/AdminController.php

public function getAllVisits(Request $request)
{
    $query = StudentHomeVisit::with([
        'student',
        'zone',
        'participants',
        'images',
        'creator'
    ]);

    // Apply filters
    if ($request->startDate) {
        $query->where('visit_date', '>=', $request->startDate);
    }

    if ($request->endDate) {
        $query->where('visit_date', '<=', $request->endDate);
    }

    if ($request->status) {
        $query->where('visit_status', $request->status);
    }

    if ($request->zoneId) {
        $query->where('zone_id', $request->zoneId);
    }

    // Count images
    $query->withCount('images');

    return $query->orderBy('visit_date', 'desc')->get();
}

public function exportToExcel(Request $request)
{
    // Use Laravel Excel
    return Excel::download(
        new HomeVisitsExport($request->visits),
        'home-visits-' . now()->format('Y-m-d') . '.xlsx'
    );
}

public function exportAllToPDF(Request $request)
{
    // Use Laravel PDF (DomPDF/TCPDF)
    $visits = StudentHomeVisit::with([...])->find($request->visits);

    $pdf = PDF::loadView('reports.home-visits-summary', [
        'visits' => $visits,
        'filters' => $request->filters
    ]);

    return $pdf->download('home-visits-summary.pdf');
}

public function downloadReport($visitId)
{
    $visit = StudentHomeVisit::with([...])->findOrFail($visitId);

    $pdf = PDF::loadView('reports.home-visit-detail', [
        'visit' => $visit
    ]);

    return $pdf->download("home-visit-{$visitId}.pdf");
}
```
